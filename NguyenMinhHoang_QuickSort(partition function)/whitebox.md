private static <T extends Comparable> int partition(T[] array, int left, int right) { 
    int mid = (left + right) / 2;               //1 
    T pivot = array[mid];                       //2

    while(left <= right) {                      //3
        while(less(array[left], pivot)){        //4
            ++left;                             //5
        }
        while(less(pivot, array[right])) {      //6
            --right;                            //7
        } 
        if (left <= right) {                    //8
            swap(array, left, right);           //9
            ++left;                             //10
            --right;                            //11
        }
    }
    return left;                                //12
}
